2) Supabase SQL schema (paste-ready)

Paste into Supabase SQL Editor. This includes:

uuid PKs

basic constraints

contact token for link-based access

timestamps

status enums as text checks (simple for v1)

-- Enable UUID generation
create extension if not exists "pgcrypto";

-- USERS
create table if not exists users (
  id uuid primary key default gen_random_uuid(),
  name text not null,
  phone text,
  timezone text default 'Australia/Melbourne',
  created_at timestamptz not null default now()
);

-- SETTINGS (1 row per user)
create table if not exists settings (
  user_id uuid primary key references users(id) on delete cascade,
  checkin_interval_hours int not null default 24,
  grace_minutes int not null default 15,
  location_mode text not null default 'off'
    check (location_mode in ('off', 'emergency_only', 'on_shift_only')),
  pause_until timestamptz,
  updated_at timestamptz not null default now()
);

-- CONTACTS (Emergency contacts)
create table if not exists contacts (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null references users(id) on delete cascade,
  name text not null,
  phone text not null,
  priority int not null check (priority in (1,2)),
  can_view_location boolean not null default true,
  created_at timestamptz not null default now(),
  unique (user_id, priority)
);

-- CONTACT TOKENS (for /c/[token] access)
create table if not exists contact_tokens (
  id uuid primary key default gen_random_uuid(),
  contact_id uuid not null references contacts(id) on delete cascade,
  token text not null unique,
  revoked boolean not null default false,
  created_at timestamptz not null default now()
);

-- CHECKINS
create table if not exists checkins (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null references users(id) on delete cascade,
  method text not null default 'button' check (method in ('button')),
  created_at timestamptz not null default now()
);

-- INCIDENTS
create table if not exists incidents (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null references users(id) on delete cascade,
  status text not null check (status in ('open', 'paused', 'resolved')),
  reason text not null check (reason in ('missed_checkin', 'sos', 'test')),
  started_at timestamptz not null default now(),
  resolved_at timestamptz,
  handled_by_contact_id uuid references contacts(id),
  next_action_at timestamptz
);

create index if not exists incidents_user_status_idx on incidents(user_id, status);
create index if not exists incidents_next_action_idx on incidents(next_action_at);

-- LOCATION SESSIONS (store minimal last known point)
create table if not exists location_sessions (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null references users(id) on delete cascade,
  incident_id uuid references incidents(id) on delete set null,
  type text not null check (type in ('emergency', 'shift')),
  active boolean not null default true,
  expires_at timestamptz not null,
  last_lat double precision,
  last_lng double precision,
  last_accuracy double precision,
  last_timestamp timestamptz,
  updated_at timestamptz not null default now()
);

create index if not exists location_sessions_user_active_idx on location_sessions(user_id, active);

-- OPTIONAL: helper view to get latest check-in time quickly
create or replace view user_last_checkin as
select
  u.id as user_id,
  max(c.created_at) as last_checkin_at
from users u
left join checkins c on c.user_id = u.id
group by u.id;

-- OPTIONAL: updated_at trigger for settings/location_sessions
create or replace function set_updated_at()
returns trigger as $$
begin
  new.updated_at = now();
  return new;
end;
$$ language plpgsql;

drop trigger if exists trg_settings_updated_at on settings;
create trigger trg_settings_updated_at
before update on settings
for each row execute function set_updated_at();

drop trigger if exists trg_location_sessions_updated_at on location_sessions;
create trigger trg_location_sessions_updated_at
before update on location_sessions
for each row execute function set_updated_at();

Scheduler approach on Replit (simple)

Because Replit doesn’t guarantee background jobs, do v1 like this:

Create an endpoint: GET /api/cron/tick

It runs “check overdue users” + “process next_action_at”

In testing you manually hit it

Later you attach an uptime/cron service (or Supabase scheduled functions)

Add this file:
